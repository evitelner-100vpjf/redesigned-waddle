import express from "express";
import cors from "cors";
import Anthropic from "@anthropic-ai/sdk";

const app = express();
const PORT = 5000;
const BEARER_TOKEN = "my_super_secret_123";

const ALLOWED_MODELS = [
  "claude-opus-4-6",
  "claude-opus-4-5",
  "claude-opus-4-1",
  "claude-sonnet-4-6",
  "claude-sonnet-4-5",
  "claude-haiku-4-5",
];
const EXTENDED_CONTEXT_MODELS = new Set([
  "claude-opus-4-6",
  "claude-sonnet-4-6",
  "claude-sonnet-4-5",
]);
const DEFAULT_MODEL = "claude-sonnet-4-6";
const CONTEXT_1M_BETA = "context-1m-2025-08-07";

app.use(cors());
app.use(express.json({ limit: "50mb" }));

const anthropic = new Anthropic({
  apiKey: process.env.AI_INTEGRATIONS_ANTHROPIC_API_KEY,
  baseURL: process.env.AI_INTEGRATIONS_ANTHROPIC_BASE_URL,
});

function authMiddleware(req, res, next) {
  const authHeader = req.headers.authorization;
  if (!authHeader || !authHeader.startsWith("Bearer ")) {
    return res.status(401).json({ error: "Missing or invalid Authorization header" });
  }
  if (authHeader.split(" ")[1] !== BEARER_TOKEN) {
    return res.status(403).json({ error: "Invalid token" });
  }
  next();
}

function resolveModel(model) {
  return model && ALLOWED_MODELS.includes(model) ? model : DEFAULT_MODEL;
}

// ── /api/chat (legacy) ────────────────────────────────────────────────────────
app.post("/api/chat", authMiddleware, async (req, res) => {
  const { message, model, extended_context } = req.body;
  if (!message || typeof message !== "string")
    return res.status(400).json({ error: "Field 'message' is required and must be a string" });

  const selectedModel = resolveModel(model);

  if (extended_context && !EXTENDED_CONTEXT_MODELS.has(selectedModel))
    return res.status(400).json({ error: `Model '${selectedModel}' does not support 1M context.` });

  try {
    const createParams = {
      model: selectedModel,
      max_tokens: 8192,
      system: "Ты — Claude, самая последняя и мощная модель от Anthropic. Твои знания актуальны на 2026 год.",
      messages: [{ role: "user", content: message }],
    };

    const response = extended_context
      ? await anthropic.beta.messages.create({ ...createParams, betas: [CONTEXT_1M_BETA] })
      : await anthropic.messages.create(createParams);

    const text = response.content
      .filter((b) => b.type === "text")
      .map((b) => b.text)
      .join("");

    res.json({
      reply: text,
      used_model: response.model,
      context_window: extended_context ? "1M tokens" : "200K tokens",
    });
  } catch (err) {
    console.error("Anthropic API error:", err.message);
    res.status(500).json({ error: err.message });
  }
});

// ── /v1/chat/completions — OpenAI format + tool calls + streaming ─────────────
app.post("/v1/chat/completions", authMiddleware, async (req, res) => {
  const { messages, model, tools, tool_choice, max_tokens, stream } = req.body;

  const selectedModel = resolveModel(model);
  const anthropicMessages = [];
  let systemPrompt = undefined;

  for (const msg of messages ?? []) {
    if (msg.role === "system") {
      systemPrompt = msg.content;
      continue;
    }
    // tool result
    if (msg.role === "tool") {
      const block = {
        type: "tool_result",
        tool_use_id: msg.tool_call_id,
        content:
          typeof msg.content === "string"
            ? msg.content
            : JSON.stringify(msg.content),
      };
      const last = anthropicMessages[anthropicMessages.length - 1];
      if (last && last.role === "user" && Array.isArray(last.content)) {
        last.content.push(block);
      } else {
        anthropicMessages.push({ role: "user", content: [block] });
      }
      continue;
    }
    // assistant with tool_calls
    if (msg.role === "assistant" && msg.tool_calls && msg.tool_calls.length) {
      const content = [];
      if (msg.content) content.push({ type: "text", text: msg.content });
      for (const tc of msg.tool_calls) {
        content.push({
          type: "tool_use",
          id: tc.id,
          name: tc.function.name,
          input: JSON.parse(tc.function.arguments || "{}"),
        });
      }
      anthropicMessages.push({ role: "assistant", content });
      continue;
    }
    anthropicMessages.push({ role: msg.role, content: msg.content });
  }

  const anthropicTools =
    tools && tools.length
      ? tools.map((t) => ({
          name: t.function.name,
          description: t.function.description || "",
          input_schema: t.function.parameters,
        }))
      : undefined;

  const useExtended = EXTENDED_CONTEXT_MODELS.has(selectedModel);

  const params = {
    model: selectedModel,
    max_tokens: max_tokens ?? 16000,
    messages: anthropicMessages,
    ...(systemPrompt ? { system: systemPrompt } : {}),
    ...(anthropicTools ? { tools: anthropicTools } : {}),
  };

  // SSE streaming — когда клиент просит stream: true
  if (stream) {
    res.setHeader("Content-Type", "text/event-stream");
    res.setHeader("Cache-Control", "no-cache");
    res.setHeader("Connection", "keep-alive");

    try {
      const streamObj = useExtended
        ? anthropic.beta.messages.stream({ ...params, betas: [CONTEXT_1M_BETA] })
        : anthropic.messages.stream(params);

      let toolCalls = {};

      streamObj.on("text", (text) => {
        const chunk = {
          id: "chatcmpl-stream",
          object: "chat.completion.chunk",
          created: Math.floor(Date.now() / 1000),
          model: selectedModel,
          choices: [{ index: 0, delta: { content: text }, finish_reason: null }],
        };
        res.write(`data: ${JSON.stringify(chunk)}\n\n`);
      });

      streamObj.on("inputJson", (delta, snapshot, info) => {
        const idx = info.index ?? 0;
        if (!toolCalls[idx]) {
          toolCalls[idx] = { id: info.id, name: info.name, arguments: "" };
          const chunk = {
            id: "chatcmpl-stream",
            object: "chat.completion.chunk",
            created: Math.floor(Date.now() / 1000),
            model: selectedModel,
            choices: [
              {
                index: 0,
                delta: {
                  tool_calls: [
                    { index: idx, id: info.id, type: "function", function: { name: info.name, arguments: "" } },
                  ],
                },
                finish_reason: null,
              },
            ],
          };
          res.write(`data: ${JSON.stringify(chunk)}\n\n`);
        }
        toolCalls[idx].arguments += delta;
        const chunk = {
          id: "chatcmpl-stream",
          object: "chat.completion.chunk",
          created: Math.floor(Date.now() / 1000),
          model: selectedModel,
          choices: [
            {
              index: 0,
              delta: { tool_calls: [{ index: idx, function: { arguments: delta } }] },
              finish_reason: null,
            },
          ],
        };
        res.write(`data: ${JSON.stringify(chunk)}\n\n`);
      });

      const finalMsg = await streamObj.finalMessage();
      const finish_reason = finalMsg.stop_reason === "tool_use" ? "tool_calls" : "stop";

      const doneChunk = {
        id: "chatcmpl-stream",
        object: "chat.completion.chunk",
        created: Math.floor(Date.now() / 1000),
        model: finalMsg.model,
        choices: [{ index: 0, delta: {}, finish_reason }],
      };
      res.write(`data: ${JSON.stringify(doneChunk)}\n\n`);
      res.write("data: [DONE]\n\n");
      res.end();
    } catch (err) {
      console.error("Streaming error:", err.message);
      res.write(`data: ${JSON.stringify({ error: err.message })}\n\n`);
      res.end();
    }
    return;
  }

  // Non-streaming — используем .stream().finalMessage() внутри,
  // чтобы Anthropic не отклонял долгие операции
  try {
    const streamObj = useExtended
      ? anthropic.beta.messages.stream({ ...params, betas: [CONTEXT_1M_BETA] })
      : anthropic.messages.stream(params);

    const response = await streamObj.finalMessage();

    let content = null;
    const toolCalls = [];

    for (const block of response.content) {
      if (block.type === "text") {
        content = (content ?? "") + block.text;
      } else if (block.type === "tool_use") {
        toolCalls.push({
          id: block.id,
          type: "function",
          function: {
            name: block.name,
            arguments: JSON.stringify(block.input),
          },
        });
      }
    }

    const finish_reason = response.stop_reason === "tool_use" ? "tool_calls" : "stop";

    res.json({
      id: `chatcmpl-${response.id}`,
      object: "chat.completion",
      created: Math.floor(Date.now() / 1000),
      model: response.model,
      choices: [
        {
          index: 0,
          message: {
            role: "assistant",
            content: content,
            ...(toolCalls.length ? { tool_calls: toolCalls } : {}),
          },
          finish_reason,
          logprobs: null,
        },
      ],
      usage: {
        prompt_tokens: response.usage.input_tokens,
        completion_tokens: response.usage.output_tokens,
        total_tokens: response.usage.input_tokens + response.usage.output_tokens,
      },
    });
  } catch (err) {
    console.error("Anthropic API error:", err.message);
    res.status(500).json({ error: err.message });
  }
});

app.get("/", (req, res) => {
  res.json({
    status: "running",
    endpoints: ["POST /api/chat", "POST /v1/chat/completions"],
  });
});

app.listen(PORT, "0.0.0.0", () =>
  console.log(`Server running on port ${PORT}`)
);
